#### 1. 에러 처리

오류가 있는 `HTTP` 응답을 클라이언트에 반환하려면 `HTTPException`를 사용한다.

```py
from fastapi import FastAPI, HTTPException

app = FastAPI()

items = {"foo": "The Foo Wrestlers"}


@app.get("/items/{item_id}")
async def read_item(item_id: str):
    # 아이템이 딕셔너리에 없는 경우 HTTP 404 예외를 발생시킴
    if item_id not in items:
        # Python의 예외이기 때문에 return이 아니라 raise.
        raise HTTPException(status_code=404, detail="Item not found")
    return {"item": items[item_id]}


## 맞춤 헤더 추가
# HTTP 오류에 사용자 정의 헤더를 추가하는 것이 유용한 몇 가지 상황이 있다.
@app.get("/items-header/{item_id}")
async def read_item_header(item_id: str):
    # 아이템이 딕셔너리에 없는 경우 HTTP 404 예외를 발생시키고, 추가로 헤더를 지정하여 반환.
    if item_id not in items:
        raise HTTPException(
            status_code=404,
            detail="Item not found",
            headers={"X-Error": "There goes my error"},  # 사용자 정의 헤더 추가
        )
    return {"item": items[item_id]}
```

1. 사용자 정의 예외 처리기 설치
`@app.exception_handler()`를 사용하여 사용자 정의 예외 처리기를 추가할 수 있다.

```py
from fastapi import FastAPI, Request
from fastapi.responses import JSONResponse

# 사용자 정의 예외 클래스 정의
class UnicornException(Exception):
    def __init__(self, name: str):
        self.name = name


app = FastAPI()


@app.exception_handler(UnicornException)
async def unicorn_exception_handler(request: Request, exc: UnicornException):
    # HTTP 상태 코드 418 (I'm a teapot)를 반환하고 사용자 지정 메시지를 JSON 형식으로 반환
    return JSONResponse(
        status_code=418,
        content={"message": f"Oops! {exc.name} did something. There goes a rainbow..."},
    )


@app.get("/unicorns/{name}")
async def read_unicorn(name: str):
    # 만약 요청된 name이 "yolo"인 경우 UnicornException을 발생시킴
    if name == "yolo":
        raise UnicornException(name=name)
    return {"unicorn_name": name}

```

2. 예외 처리기 재정의
`FastAPI` 에는 몇 가지 기본 예외 처리기가 있다.

```py
from fastapi import FastAPI, HTTPException
from fastapi.exceptions import RequestValidationError
from fastapi.responses import PlainTextResponse
from starlette.exceptions import HTTPException as StarletteHTTPException

app = FastAPI()


# Starlette의 HTTPException을 처리하는 exception handler 함수
@app.exception_handler(StarletteHTTPException)
async def http_exception_handler(request, exc):
    # HTTP 예외의 상세 내용을 텍스트 응답으로 반환하고 상태 코드를 유지
    return PlainTextResponse(str(exc.detail), status_code=exc.status_code)


# RequestValidationError을 처리하는 exception handler 함수
@app.exception_handler(RequestValidationError)
async def validation_exception_handler(request, exc):
    # 요청 유효성 검사 예외의 내용을 텍스트 응답으로 반환하고 상태 코드 400을 반환
    return PlainTextResponse(str(exc), status_code=400)


@app.get("/items/{item_id}")
async def read_item(item_id: int):
    # 만약 item_id가 3인 경우 HTTP 418 예외 발생
    if item_id == 3:
        raise HTTPException(status_code=418, detail="Nope! I don't like 3.")
    return {"item_id": item_id}
```

3. 동일한 기본 예외 처리기와 함께 예외를 사용하려는 경우 `fastapi.exception_handler`에서 기본 예외 처리기를 가져와 다시 사용할 수 있다.

```py
from fastapi import FastAPI, HTTPException
from fastapi.exception_handlers import (
    http_exception_handler,
    request_validation_exception_handler,
)
from fastapi.exceptions import RequestValidationError
from starlette.exceptions import HTTPException as StarletteHTTPException

app = FastAPI()

@app.exception_handler(StarletteHTTPException)
async def custom_http_exception_handler(request, exc):
    # 요청 유효성 검사 예외가 발생했을 때 출력하고 기본 처리기에 예외를 전달하여 처리
    print(f"OMG! An HTTP error!: {repr(exc)}")
    return await http_exception_handler(request, exc)

@app.exception_handler(RequestValidationError)
async def validation_exception_handler(request, exc):
    # 요청 유효성 검사 예외가 발생했을 때 출력하고 기본 처리기에 예외를 전달하여 처리
    print(f"OMG! The client sent invalid data!: {exc}")
    return await request_validation_exception_handler(request, exc)

@app.get("/items/{item_id}")
async def read_item(item_id: int):
    if item_id == 3:
        raise HTTPException(status_code=418, detail="Nope! I don't like 3.")
    return {"item_id": item_id}
```

#### 2. 경로 작동 설정

경로 작동 데코레이터를 설정하기 위해서 전달할수 있는 몇 가지 매개변수가 있다.

1. 응답 상태 코드
경로 작동의 응답에 사용될 (`HTTP`) `status_code`를 정의할수 있습니다.

```py
# Item 모델을 반환하고, 상태 코드는 201 Created를 반환
@app.post("/items/", response_model=Item, status_code=status.HTTP_201_CREATED)
async def create_item(item: Item):
    return item

# 404와 같은 int형 코드를 직접 전달할수 있다.
# 각 코드의 의미를 모른다면, status에 있는 단축 상수들을 사용할수 있다.
```

2. 태그
`str`로 구성된 `list`와 함께 매개변수 `tags`를 전달하여, 경로 작동에 태그를 추가할 수 있다.

```py
#  'items' 태그 지정
@app.post("/items/", response_model=Item, tags=["items"])
async def create_item(item: Item):
    return item


#  'items' 태그 지정
@app.get("/items/", tags=["items"])
async def read_items():
    return [{"name": "Foo", "price": 42}]


#  'users' 태그 지정
@app.get("/users/", tags=["users"])
async def read_users():
    return [{"username": "johndoe"}]

# 전달된 태그들은 OpenAPI의 스키마에 추가되며, 자동 문서 인터페이스에서 사용된다.
```

3. 요약 및 설명
`summary` 및 `description`을 추가할 수 있다.

```py
#  요약(summary)과 설명(description)은 문서화를 위해 제공
@app.post(
    "/items/",
    response_model=Item,
    summary="Create an item",
    description="Create an item with all the information, name, description, price, tax and a set of unique tags",
)
async def create_item(item: Item):
    return item
```

4. 독스트링
`docstring`에서 경로 연산 설명을 선언하면 `FastAPI`가 이를 읽어온다.

```py
# 함수의 주석에는 엔드포인트의 동작과 관련된 상세 설명이 포함되어 있다.
@app.post(
    "/items/",
    response_model=Item,
    summary="Create an item",
    response_description="The created item", # 응답에 대한 설명을 나타내는 문자열
)
async def create_item(item: Item):
    """
    Create an item with all the information:

    - **name**: each item must have a name
    - **description**: a long description
    - **price**: required
    - **tax**: if the item doesn't have tax, you can omit this
    - **tags**: a set of unique tag strings for this item
    """
    return item
```

5. 경로 작업 지원 중단
경로 작업을 제거하지 않고 비활성화된 것으로 표시해야 하는 경우에는 `deprecated` 매개 변수를 전달한다.

```py
from fastapi import FastAPI

app = FastAPI()


@app.get("/items/", tags=["items"])
async def read_items():
    return [{"name": "Foo", "price": 42}]


@app.get("/users/", tags=["users"])
async def read_users():
    return [{"username": "johndoe"}]

# deprecated=True를 설정하여, 이 엔드포인트는 더 이상 사용되지 않을 것임을 나타냄
@app.get("/elements/", tags=["items"], deprecated=True)
async def read_elements():
    return [{"item_id": "Foo"}]
```

#### 3. 본문 업데이트

```py
# PUT으로 업데이트
@app.put("/items/{item_id}", response_model=Item)
async def update_item(item_id: str, item: Item):
    update_item_encoded = jsonable_encoder(item)
    items[item_id] = update_item_encoded
    return update_item_encoded

# PATCH로 부분 업데이트
@app.patch("/items/{item_id}", response_model=Item)
async def update_item(item_id: str, item: Item):
    stored_item_data = items[item_id]
    stored_item_model = Item(**stored_item_data)
    # 업데이트할 데이터를 추출
    update_data = item.dict(exclude_unset=True)
    # `copy` 메서드를 사용하여 업데이트된 데이터로 `Item` 모델의 새 인스턴스를 생성
    updated_item = stored_item_model.copy(update=update_data)
    # 업데이트된 항목을 데이터 저장소에 다시 저장
    items[item_id] = jsonable_encoder(updated_item)
    return updated_item
```
